## Array 데이터 추가

```html
<html>
    <head>
        <title>

        </title>
    </head>
    <body>
        <script type="text/javascript">
            var arr = ['1','2','3','4','5'];
            
            alert(arr);

            arr.push('6');
            //배열의 맨 뒤에 데이터 추가
            alert(arr);

            arr = arr.concat('7','8');
            //배열의 맨 뒤에 복수의 데이터 추가
            alert(arr);

            arr.unshift('0');
            //배열의 앞에 데이터 추가
            alert(arr);

            arr.splice(2,0,'B');
            //배열의 첫번째 인자 수 다음에 데이터 추가, 두번째 인자에 해당하는 원소의 숫자만큼의 값을 배열로부터 제거한 후에 리턴
            alert(arr);

            arr.shift();
            //배열의 첫번째 데이터 삭제
            alert(arr);

            arr.pop();
            //배열의 마지막 데이터 삭제
            alert(arr);

            arr.reverse();
            //배열 역순으로 정렬
            alert(arr);
            
            arr.sort();
            //배열 정렬
            alert(arr);

            document.write('배열의 갯수는 = ' + arr.length);
            </script>
    </body>
</html>
```
